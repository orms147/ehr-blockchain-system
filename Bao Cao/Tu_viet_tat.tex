% ============== THUẬT NGỮ chuyên ngành ==============

\newglossaryentry{blockchain}{
    name=Blockchain,
    description={Sổ cái phân tán liên kết các khối dữ liệu bằng hàm băm mật mã, không thể chỉnh sửa sau khi ghi nhận.}
}

\newglossaryentry{smartcontract}{
    name=Smart Contract,
    description={Hợp đồng thông minh — đoạn mã chạy trên máy ảo EVM, thực thi tự động khi điều kiện đạt được.}
}

\newglossaryentry{consent}{
    name=Consent,
    description={Sự đồng thuận của bệnh nhân cho phép một bên thứ ba (bác sĩ, tổ chức) truy cập hồ sơ y tế.}
}

\newglossaryentry{delegation}{
    name=Delegation,
    description={Cơ chế cho phép bác sĩ được uỷ quyền chuyển tiếp quyền truy cập cho bác sĩ khác (nếu allowDelegate=true).}
}

\newglossaryentry{blindmailbox}{
    name=Blind Mailbox,
    description={Mô hình kiến trúc trong đó máy chủ chỉ lưu trữ dữ liệu đã mã hoá mà không có khoá giải mã.}
}

\newglossaryentry{trustedcontact}{
    name=Trusted Contact,
    description={Người thân tin cậy được bệnh nhân uỷ quyền truy cập hồ sơ y tế trong tình huống khẩn cấp.}
}

\newglossaryentry{rollup}{
    name=Rollup,
    description={Giải pháp Layer 2 mở rộng Ethereum, gộp nhiều giao dịch off-chain rồi commit lên L1 để giảm phí gas.}
}

\newglossaryentry{subgraph}{
    name=Subgraph,
    description={Cấu trúc index trên giao thức The Graph, giúp truy vấn sự kiện blockchain nhanh hơn so với poll RPC trực tiếp.}
}

\newglossaryentry{relayer}{
    name=Relayer,
    description={Dịch vụ trung gian thay mặt người dùng gửi giao dịch lên blockchain và thanh toán phí gas, dựa trên chữ ký EIP-712.}
}
