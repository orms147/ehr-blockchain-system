% Cấu hình listings — formatting cho code snippet (Solidity, JS, TS, JSON, bash).
% Được include trong quyen.tex line 62.

\usepackage{xcolor}

\definecolor{codegreen}{rgb}{0,0.6,0}
\definecolor{codegray}{rgb}{0.5,0.5,0.5}
\definecolor{codepurple}{rgb}{0.58,0,0.82}
\definecolor{codeorange}{rgb}{1.0,0.49,0}
\definecolor{backcolour}{rgb}{0.96,0.96,0.96}

\lstdefinestyle{commonstyle}{
    backgroundcolor=\color{backcolour},
    commentstyle=\color{codegreen},
    keywordstyle=\color{codepurple}\bfseries,
    numberstyle=\tiny\color{codegray},
    stringstyle=\color{codeorange},
    basicstyle=\ttfamily\footnotesize,
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    keepspaces=true,
    numbers=left,
    numbersep=5pt,
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
    frame=single,
    rulecolor=\color{codegray}
}

% Solidity language definition (không có sẵn trong listings).
\lstdefinelanguage{Solidity}{
    keywords={contract, function, modifier, event, struct, mapping, address, uint, uint256, uint40, uint8, bool, bytes32, string, public, private, internal, external, view, pure, payable, returns, return, if, else, for, while, require, revert, emit, memory, storage, calldata, constant, immutable, override, virtual, abstract, interface, library, using, import, pragma, solidity, msg, sender, value, block, timestamp, this, new, delete, indexed, anonymous, true, false},
    keywordstyle=\color{codepurple}\bfseries,
    sensitive=true,
    comment=[l]{//},
    morecomment=[s]{/*}{*/},
    string=[b]"
}

\lstdefinestyle{solidity}{
    style=commonstyle,
    language=Solidity
}

\lstdefinestyle{javascript}{
    style=commonstyle,
    language=JavaScript
}

\lstdefinestyle{typescript}{
    style=commonstyle,
    language=JavaScript,
    morekeywords={interface, type, as, let, const, async, await, export, import, from, of}
}

\lstdefinestyle{json}{
    style=commonstyle,
    language=,
    keywords={true, false, null}
}

\lstdefinestyle{bash}{
    style=commonstyle,
    language=bash
}

\lstset{style=commonstyle}
